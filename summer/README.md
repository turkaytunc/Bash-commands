```bash
# "<" input stream, ">" output stream
echo "Hello World" > hello.txt
---
# copy file content to another file using cat and input-output stream
cat < hello.txt > newFile.txt
```

```bash
# ">>" append
cat < hello.txt >> newFile.txt
```
